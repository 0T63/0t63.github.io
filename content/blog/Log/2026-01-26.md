```bash
# 新增所有修改的檔案到暫存區
git add .

# 查看當前檔案狀態
git status

# 提交修改，"XXXXX" 可替換為提交訊息
git commit -m "XXXXX"

# 切換分支
git switch

# 查看分支列表
git branch

# 新增遠端倉庫
git remote add <名稱> <倉庫地址>

# 查看遠端倉庫
git remote -v

# 查看提交歷史
git log

# 拉取遠端修改
git pull

# 推送本地修改到遠端
git push

# 克隆遠端倉庫
git clone <倉庫地址>

# 新建分支並切換
git switch -c <分支名>


```
